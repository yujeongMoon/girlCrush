create table customer(
Id number(10,0) not null primary key,
EMAIL VARCHAR2(100) NOT NULL,
Password varchar2(100) not null,
travel_id number(10,0)
);

CREATE SEQUENCE ID_SEQ
START WITH 1
INCREMENT BY 1;

CREATE SEQUENCE ID_SEQ_TRAVEL
START WITH 1
INCREMENT BY 1;

CREATE SEQUENCE ID_SEQ_NOTICE
START WITH 1
INCREMENT BY 1;

Create table travel_board(
ID NUMBER(10,0) NOT NULL PRIMARY KEY,
WRITER VARCHAR2(100) DEFAULT NULL,
TITLE VARCHAR2(255) DEFAULT NULL,
CONTENT VARCHAR2(4000) DEFAULT NULL,
REG_DATE DATE DEFAULT NULL,
HIT_COUNT NUMBER(10,0) DEFAULT 0
);

Create table NOTICE_BOARD(
ID NUMBER(10,0) NOT NULL PRIMARY KEY,
WRITER VARCHAR2(100) DEFAULT NULL,
TITLE VARCHAR2(255) DEFAULT NULL,
CONTENT VARCHAR2(4000) DEFAULT NULL,
REG_DATE DATE DEFAULT NULL,
HIT_COUNT NUMBER(10,0) DEFAULT 0
);

Create table PAYMENT_BOARD(
ID NUMBER(10,0) NOT NULL PRIMARY KEY,
NAME VARCHAR2(20) NOT NULL,
EMAIL VARCHAR2(30) NOT NULL,
ADDRESS VARCHAR2(30) NOT NULL,
NAME_ON_CATD VARCHAR2(20) NOT NULL,
CREDITCARD_NUMBER VARCHAR2(30) NOT NULL,
EXP_MONTH NUMBER NOT NULL,
EXP_YEAR NUMBER NOT NULL,
CVV NUMBER NOT NULL
);

CREATE SEQUENCE ID_SEQ_PAYMENT
START WITH 1
INCREMENT BY 1;
